#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int secretNumber, guess, attempts = 0;
    int maxAttempts;
    int choice;

    // Seed random number generator
    srand(time(0));
    secretNumber = rand() % 100 + 1;  // 1 to 100

    // Difficulty selection
    printf("===== NUMBER GUESSING GAME =====\n");
    printf("Select Difficulty Level:\n");
    printf("1. Easy (15 guesses)\n");
    printf("2. Medium (10 guesses)\n");
    printf("3. Hard (5 guesses)\n");
    printf("Enter your choice: ");
    scanf("%d", &choice);

    switch (choice) {
        case 1:
            maxAttempts = 15;
            break;
        case 2:
            maxAttempts = 10;
            break;
        case 3:
            maxAttempts = 5;
            break;
        default:
            printf("Invalid choice. Defaulting to Medium.\n");
            maxAttempts = 10;
    }

    printf("\nI have selected a number between 1 and 100.\n");
    printf("Try to guess it!\n");

    // Guessing loop
    while (attempts < maxAttempts) {
        printf("\nEnter your guess: ");
        scanf("%d", &guess);
        attempts++;

        if (guess > secretNumber) {
            printf("Too High!");
        } else if (guess < secretNumber) {
            printf("Too Low!");
        } else {
            printf("\nðŸŽ‰ Congratulations! You guessed the number in %d attempts.\n", attempts);
            return 0;
        }
    }

    // If user runs out of attempts
    printf("\nâŒ You've used all %d attempts.\n", maxAttempts);
    printf("The correct number was: %d\n", secretNumber);

    return 0;
}
