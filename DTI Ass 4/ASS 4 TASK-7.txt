#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// Function to print ASCII dice face
void printDice(int num) {
    switch (num) {
        case 1:
            printf("┌─────┐\n");
            printf("│     │\n");
            printf("│  •  │\n");
            printf("│     │\n");
            printf("└─────┘\n");
            break;
        case 2:
            printf("┌─────┐\n");
            printf("│ •   │\n");
            printf("│     │\n");
            printf("│   • │\n");
            printf("└─────┘\n");
            break;
        case 3:
            printf("┌─────┐\n");
            printf("│ •   │\n");
            printf("│  •  │\n");
            printf("│   • │\n");
            printf("└─────┘\n");
            break;
        case 4:
            printf("┌─────┐\n");
            printf("│ • • │\n");
            printf("│     │\n");
            printf("│ • • │\n");
            printf("└─────┘\n");
            break;
        case 5:
            printf("┌─────┐\n");
            printf("│ • • │\n");
            printf("│  •  │\n");
            printf("│ • • │\n");
            printf("└─────┘\n");
            break;
        case 6:
            printf("┌─────┐\n");
            printf("│ • • │\n");
            printf("│ • • │\n");
            printf("│ • • │\n");
            printf("└─────┘\n");
            break;
    }
}

int main() {
    int numDice, roll;
    char choice;
    int sum, i;
    double avg;

    srand(time(0)); // Seed random number generator

    printf("===== DICE ROLLING SIMULATOR =====\n");

    do {
        printf("\nEnter number of dice to roll: ");
        scanf("%d", &numDice);

        if (numDice <= 0) {
            printf("Please enter a positive number!\n");
            continue;
        }

        sum = 0;

        printf("\nRolling %d dice...\n", numDice);
        for (i = 0; i < numDice; i++) {
            roll = rand() % 6 + 1; // 1 to 6
            printf("Dice %d: %d\n", i + 1, roll);
            printDice(roll); // ASCII dice
            sum += roll;
        }

        avg = (double)sum / numDice;
        printf("Total Sum: %d\n", sum);
        printf("Average: %.2lf\n", avg);

        printf("\nRoll again? (y/n): ");
        scanf(" %c", &choice);

    } while (choice == 'y' || choice == 'Y');

    printf("\nThanks for playing!\n");
    return 0;
}
