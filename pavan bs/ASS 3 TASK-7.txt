#include <stdio.h>

int main() {
    char str[100], rev[100];
    int i, length = 0, vowels = 0, isPalindrome = 1;

    // Input string
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);

    // Find length (excluding newline)
    for (i = 0; str[i] != '\0'; i++) {
        if (str[i] == '\n')
            break;
        length++;
    }

    // Reverse string manually
    for (i = 0; i < length; i++) {
        rev[i] = str[length - i - 1];
    }
    rev[length] = '\0';

    // Count vowels
    for (i = 0; i < length; i++) {
        char ch = str[i];
        if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
            ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U') {
            vowels++;
        }
    }

    // Check palindrome
    for (i = 0; i < length; i++) {
        if (str[i] != rev[i]) {
            isPalindrome = 0;
            break;
        }
    }

    // Output results
    printf("\nReversed String: %s", rev);
    printf("\nVowel Count: %d", vowels);

    if (isPalindrome)
        printf("\nThe string is a palindrome.\n");
    else
        printf("\nThe string is not a palindrome.\n");

    return 0;
}
