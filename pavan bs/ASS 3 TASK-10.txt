#include <stdio.h>

// Function to swap two numbers using pointers
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x = 10, y = 20;
    int arr[5] = {1, 2, 3, 4, 5};
    int *ptr;
    int i;

    // Print values and addresses before swap
    printf("Before Swap:\n");
    printf("x = %d, Address of x = %p\n", x, (void*)&x);
    printf("y = %d, Address of y = %p\n", y, (void*)&y);

    // Swap using pointers
    swap(&x, &y);

    // Print values and addresses after swap
    printf("\nAfter Swap:\n");
    printf("x = %d, Address of x = %p\n", x, (void*)&x);
    printf("y = %d, Address of y = %p\n", y, (void*)&y);

    // Traverse array using pointer
    printf("\nArray elements using pointers:\n");
    ptr = arr;

    for (i = 0; i < 5; i++) {
        printf("Value: %d, Address: %p\n", *ptr, (void*)ptr);
        ptr++;   // Move to next element
    }

    return 0;
}
