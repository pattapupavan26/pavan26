#include <stdio.h>
#include <stdlib.h>

struct Account {
    int accNo;
    char name[50];
    float balance;
};

void createAccount(struct Account *a);
void deposit(struct Account *a);
void withdraw(struct Account *a);
void checkBalance(struct Account *a);
void saveToFile(struct Account *a);
int loadFromFile(struct Account *a);

int main() {
    struct Account acc;
    int choice;
    int exists = loadFromFile(&acc);

    do {
        printf("\n===== BANKING SYSTEM =====\n");
        printf("1. Create Account\n");
        printf("2. Deposit\n");
        printf("3. Withdraw\n");
        printf("4. Check Balance\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                createAccount(&acc);
                saveToFile(&acc);
                exists = 1;
                break;

            case 2:
                if (exists) {
                    deposit(&acc);
                    saveToFile(&acc);
                } else {
                    printf("No account found. Create an account first.\n");
                }
                break;

            case 3:
                if (exists) {
                    withdraw(&acc);
                    saveToFile(&acc);
                } else {
                    printf("No account found. Create an account first.\n");
                }
                break;

            case 4:
                if (exists)
                    checkBalance(&acc);
                else
                    printf("No account found.\n");
                break;

            case 5:
                printf("Thank you for using the Banking System.\n");
                break;

            default:
                printf("Invalid choice!\n");
        }

    } while (choice != 5);

    return 0;
}

/* Function Definitions */

void createAccount(struct Account *a) {
    printf("\nEnter Account Number: ");
    scanf("%d", &a->accNo);

    printf("Enter Name: ");
    scanf(" %[^\n]", a->name);

    printf("Enter Initial Balance: ");
    scanf("%f", &a->balance);

    printf("Account created successfully!\n");
}

void deposit(struct Account *a) {
    float amount;
    printf("Enter amount to deposit: ");
    scanf("%f", &amount);

    if (amount > 0) {
        a->balance += amount;
        printf("Deposit successful.\n");
    } else {
        printf("Invalid amount.\n");
    }
}

void withdraw(struct Account *a) {
    float amount;
    printf("Enter amount to withdraw: ");
    scanf("%f", &amount);

    if (amount > 0 && amount <= a->balance) {
        a->balance -= amount;
        printf("Withdrawal successful.\n");
    } else {
        printf("Insufficient balance or invalid amount.\n");
    }
}

void checkBalance(struct Account *a) {
    printf("\n--- Account Details ---\n");
    printf("Account No: %d\n", a->accNo);
    printf("Name: %s\n", a->name);
    printf("Balance: %.2f\n", a->balance);
}

void saveToFile(struct Account *a) {
    FILE *fp = fopen("bank.dat", "wb");
    if (fp == NULL) {
        printf("Error saving data.\n");
        return;
    }
    fwrite(a, sizeof(struct Account), 1, fp);
    fclose(fp);
}

int loadFromFile(struct Account *a) {
    FILE *fp = fopen("bank.dat", "rb");
    if (fp == NULL)
        return 0;

    fread(a, sizeof(struct Account), 1, fp);
    fclose(fp);
    return 1;
}
