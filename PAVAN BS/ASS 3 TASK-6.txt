#include <stdio.h>

int main() {
    int n, i;
    int sum = 0, largest, smallest;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    int arr[n];   // array size decided at runtime (not hardcoded)

    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    largest = smallest = arr[0];

    for (i = 0; i < n; i++) {
        sum += arr[i];

        if (arr[i] > largest)
            largest = arr[i];

        if (arr[i] < smallest)
            smallest = arr[i];
    }

    printf("\nSum = %d\n", sum);
    printf("Average = %.2f\n", (float)sum / n);
    printf("Largest = %d\n", largest);
    printf("Smallest = %d\n", smallest);

    return 0;
}
